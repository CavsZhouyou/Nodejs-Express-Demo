CREATE TABLE IF NOT EXISTS movie_time(
   movie_id INT UNSIGNED NOT NULL,
   show_date VARCHAR(50) NOT NULL,
   show_time VARCHAR(50) NOT NULL,
   PRIMARY KEY (movie_id,show_date,show_time)
   )ENGINE=InnoDB DEFAULT CHARSET=utf8;
